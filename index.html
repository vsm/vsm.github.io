vsmjs

Translating any knowledge into a computable form, made easy.

This page is a placeholder that will one day unite all info on VSM (see the extensive VSM-pages), and its supporting software (the vsmjs project).

The core of the vsmjs project is the vsm-box user-interface / web-component.
Check the live vsm-box demo.



Here is a minimal demo of how to use 'vsm-box' in your project.
Paste this code in a file 'test.html' and open it in your browser: 


<!doctype html>
<html>
<head>
  <script src="https://unpkg.com/vsm-dictionary-local@^2.6.0/dist/vsm-dictionary-local.min.js"></script>
  <script src="https://unpkg.com/vsm-dictionary-cacher@^1.2.0/dist/vsm-dictionary-cacher.min.js"></script>
  <script src="https://unpkg.com/vsm-box@^1.0.0/dist/vsm-box.standalone.min.js"></script>
</head>

<body>

  <vsm-box autofocus=true></vsm-box>

  <br>
  <textarea readonly></textarea>
</body>

<script>
  var vsmbox = document.getElementsByTagName('vsm-box')[0];

  vsmbox.vsmDictionary = new (VsmDictionaryCacher(VsmDictionaryLocal)) ({
    dictData: [ { id: 'http://d.org/A',  name: 'Aaa',  entries: [
      { id: 'http://d.org/A/A:01', terms: [{ str: 'aaa' }, { str: 'aab' }] },
      { id: 'http://d.org/A/A:03', terms: [{ str: 'abc' }], descr: 'example' }
    ]}]
  });  // (Or use another dictionary interface, like vsm-dictionary-bioportal).


  // Uncomment this to fill in something:
  //vsmbox.initialValue = {
  //  terms: [ {}, { str: 'aaa', classID: 'http://d.org/A/A:01', instID: null }, {} ],
  //  conns: [ { type: 'T', pos: [ 0, 1, 2 ] } ] };

  // Show changes in the textarea.
  var text = document.getElementsByTagName('textarea')[0];  text.value = '';
  function showOutput(e) { text.value = JSON.stringify(e.detail[0], null, 2) }
  vsmbox.addEventListener('change', showOutput);
</script>
</html>
